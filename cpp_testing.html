This first program is from the Google Education C++ Course. It's designed to give an introduction to IO in C++, and begin to form good practices and code style.

    // get_input.cpp: Josh Hartigan
    // Description: Illustrate usage of cin to get input.

    #include <iostream>
    using namespace std;

    int main() {
      int input_var = 0;

      do {
        cout << "enter a number (-1 = quit): ";
        // cin returns false if an input operation fails,
        // i.e. the type of input is not what is expected.
        if (!(cin >> input_var)) {
          cout << "*** Input NaN." << endl;
          cin.clear();
          cin.ignore(10000, '\n'); // chars use single quotes
          // break; // exit do-while loop
        }
        // all code after the previous if statement will
        // now be executed if input_var is an integer.
        else if (input_var != -1) {
          cout << "your number was " << input_var << endl;
        }
      } while (input_var != -1);

      cout << "goodbye." << endl;
      return 0;
    }

A useless program I wrote to practice the usage of reference compound types:

    #include 
    using namespace std;

    int main() {
      cout << endl;

      int myNumber = 22;
      int &ref = myNumber; // ref 'references' myNumber

      myNumber++;

      cout << "myNumber: " << myNumber; // ==> 23
      cout << "\nref: " << ref;         // ==> Also 23, because it's
                                        //     not a seperate variable.
      ref++; // increment to 24 - not
             // incrementing 'ref's value,
             // but the value of whatever
             // it references (myNumber)

      cout << "\nmyNumber: " << myNumber; // ==> 24 - not 23.

      cout << endl;
      return 0;
    }
